	string desc = "C++ is a beautiul! do you like it?";    
	string newDesc = "";

int main() {
	vector<string> text {"hello world", "", "bye bye"};
	for (auto it = text.begin(); it != text.end(); ++it) {
		for (auto i = (*it).begin(); i != (*it).end(); ++i) {
			*i = toupper(*i);
		}
		
	}
	
	cout << text[0] << endl;       //=> HELLO WORLD
	return 0;
}

**ex 3.10**
```
# include <iostream>
using namespace std;
// 将字符串去除标点符号
int main() {
	string desc = "C++ is a beautiul! do you like it?";    
	string newDesc = ""; 
	for (auto &c : desc) {
		if (!ispunct(c)) {
			newDesc += c;
		}
	}
	cout << newDesc << endl;
	return 0;
}
```

**ex 3.11**

using namespace std;

int main() {
	const string s = "keep out!";       //=> s 是常量
	for (auto &c : s) {
		toupper(c);                     //=> c's type is 'const char &'
	}	
	cout << s << endl;                  //=> keep out!
	return 0;
}

**ex 2.3**
```
# include <iostream>
using namespace std;

int main() {
	unsigned u = 10, u2 = 42;

	std::cout << u2 - u << std::endl;  // 32
	cout << u - u2 << endl;            // 4294967264

	int i = 10, i2 = 42;
	cout << i2 - i << endl;             // 32
	cout << i - i2 << endl;             // -32
	cout << i - u << endl;              // 0
	cout << u - i << endl;              // 0
 	return 0;
}
```

**ex 2.10**

```

int global_int;

	int local_int;
	std::string local_str;

	std::cout << "global_str: " << global_str << "end" << std::endl;
	std::cout << "global_int: " << global_int << std::endl;
	std::cout << "local_int: " << local_int << std::endl;

 	return 0;
}
```

**ex 2.14**
合法，for语句中的i变量只在for语句中有效
```
# include <iostream>

int main() {
	int i = 100, sum = 0;
	for (int i = 0; i != 10; ++i) {
		sum += i;                               // sum=> 0~10之间数的和
	}
	std::cout << i << " " << sum << std::endl;  //=> 100 45
 	return 0;
}
```

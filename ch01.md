**ex 1.11**
输出两个数间的所有整数
```
# include <iostream>

int main() {
	int beg, end;
	int num1, num2;
	std::cout << "Enter two integer: ";
	std::cin >> num1 >> num2;

	if (num1 < num2) {
		beg = num1;
		end = num2;
	}  
	
	if (num1 > num2) {
		beg = num2;
		end = num1;
	}

	if (num1 == num2){
		std::cout << num1 << std::endl;
		return 0;
	}

	while (beg <= end) {
		std::cout << beg << std::endl;
		beg++;
	}
 	return 0;
}
```
